爬取百度图片

简介

  • 什么是爬虫?

 网络爬虫也叫网络蜘蛛，它特指一类自动批量下载网络资源的程序，这是一个比较口语话的定义。
 更加专业和全面的定义是：网络爬虫是伪装成客户端与服务端进行数据交互的程序。

  • 爬虫的应用

    1. 数据采集

     大数据时代来临，数据就是核心，就是生产力，越来越多的企业开始注重收集用户数据，而爬虫技术是收集数据的一种重要手段。

    1. 搜索引擎

     爬虫也被广泛用于模拟用户操作，测试机器人，灌水机器人等。

爬虫开发的重难点

  • 爬虫的难点主要为两个方向：

 网络公共资源都是为用户准备的，为了避免被爬虫采集，服务端会设置非常多的图灵测试，阻止爬虫的恶意爬取，也即是反爬措施。爬虫开发工程师在开发爬虫时，需要解决这些反爬措施。我们在开发爬虫的过程中，有很大一部分的工作就是处理这些反爬措施。

  • 采集的速度

 大数据时代，需要巨大的数据量，动辄千万条的级别，甚至上亿条。如果采集速度跟不上，耗时过长，那么就达不到商业要求。一般我们会采取并发以及分布式来解决速度上的问题。这也是爬虫开发过程中的另外一个重心。

爬取百度图片案例

 犹记得的大一学python的时候，大概12月份左右，拿到了一个题目，通过requests与re库爬取百度图片

  • 代码大概如下：
import re
import requests
import os
#获取路径
def getPath():
    r = os.getcwd()
    root = os.path.abspath(r)
    folder = os.path.exists(root+"\\百度上爬取的图片")
    if not folder:
        print('目录不存在')
        os.mkdir(root+"\\百度上爬取的图片")
    else:
        print('目录已存在')
    print("路径名是:"+root+"\\百度上爬取的图片")
    return root+"\\百度上爬取的图片"
path = getPath()
print(path)
#下载图片
def downloadPic(html,keyword,path):
    dirName = path+"\\{}的图片".format(keyword)
    fo = os.path.exists(dirName)
    if not fo:
        os.mkdir(dirName)
        
    pic_url=re.findall('"objURL":"(.*?)",',html,re.S)
    i=1
    f = open(path+"\\爬取提示信息的日志.txt",'a',encoding = 'utf-8')
    if f.read!='':
        f.write("------------------------------------------------\n\n\n\n")
    stri = '找到关键字:'+keyword+'的图片，现在开始下载图片...'
    f.write(stri+'\n')
    print(stri)
    for each in pic_url:
        string = '正在下载第'+str(i)+'张图片，图片地址:'+str(each)
        f.write(string+"\n")
        print(string)
        try:
            pic=requests.get(each,timeout=10)
        except requests.exceptions.ConnectionError:
            str1 = '【错误】当前图片无法下载'
            f.write(str1+"\n")
            print(str1)
            continue
        dir1 = keyword+'_'+str(i)+'.jpg'
        fp = open(dirName+"\\"+dir1,'wb')
        fp.write(pic.content)
        fp.close()
        i+=1
    f.close()
def main():
    while True:
        keyword = input("Input keyword: ")
        url = 'http://image.baidu.com/search/flip?tn=baiduimage&ie=utf-8&word='+keyword+'&ct=201326592%v=flip'
        result = requests.get(url)
        path = getPath()
        downloadPic(result.text,keyword,path)
        print("Successfully downloaded")

if __name__=="__main__":
    main()
  • 遇到疑惑

 然而，刚照着敲完，发现仍然有许多地方不懂，比如http协议，还有re是什么？

 经过不断查阅资料，写成了以下的代码：

import requests,re,os
from urllib.parse import urlencode
#获取需要下载的页数和关键词
def get_PagesAndKeyword():
    pages = eval(input("Please enter the number of the page to show:"))
    keyword = input("请输入关键词列表,以中文逗号分隔example(动漫，猫):")
    return pages,keyword.split('，')
#百度链接地址加密算法解析
def baidtu_uncomplie(url):
    res = ''
    c = ['_z2C$q', '_z&e3B', 'AzdH3F']
    d = {'w': 'a', 'k': 'b', 'v': 'c', '1': 'd', 'j': 'e', 'u': 'f', '2': 'g', 'i': 'h', 't': 'i', '3': 'j', 'h': 'k',
         's': 'l', '4': 'm', 'g': 'n', '5': 'o', 'r': 'p', 'q': 'q', '6': 'r', 'f': 's', 'p': 't', '7': 'u', 'e': 'v',
         'o': 'w', '8': '1', 'd': '2', 'n': '3', '9': '4', 'c': '5', 'm': '6', '0': '7', 'b': '8', 'l': '9', 'a': '0',
         '_z2C$q': ':', '_z&e3B': '.', 'AzdH3F': '/'}
    if (url == None or 'http' in url):  # 判断地址是否有http
        return url
    else:
        j = url
        # 解码百度加密算法
        for m in c:
            j = j.replace(m, d[m])
        for char in j:
            if re.match('^[a-w\d]+$', char):    # 正则表达式
                char = d[char]
            res = res + char
        return res
#获取页面
def get_url_list(pageNumber,keyword):
    url_json_list = []
    url = 'https://image.baidu.com/search/acjson?'
    for i in range(30,30*(pageNumber+1),30):

                }
        url_json_list.append(url+urlencode(param))
    url_list = []
    for url_json in url_json_list:
        response = requests.get(url_json)
        content = response.text
        pattern = r'"objURL":"(.*?)",'
        url_list.append(re.findall(pattern,content,re.S))
    return url_list

    result_list = []
    for ls in url_list:
        for url in ls:
            result_list.append(baidtu_uncomplie(url))
  • 改进

 当我尝试运行以上的代码，却发现，时常会出现等待现象，一张图片下载许久，其他图片便无法下载，可能到最后图片还是没有下载成功。因此，我去学习了下怎么用多线程与多进程。

 改进的多进程代码如下：

from urllib.parse import urlencode
import threading
#获取需要下载的页数和关键词

def get_url_list(pageNumber,keyword):

    for i in range(30,30*(pageNumber+1),30):
        param = {
            'tn':'resultjson_com',
            'ipn':'rj',
            'ct':'201326592',
            'fp':'result',
            'queryWord':keyword,   #关键字
            'cl':'2',

def decode_url_list(url_list):

            result_list.append(baidtu_uncomplie(url))
    return result_list
#根据一个关键词创建目录
def make_dir(root_path,keyword):
    if not os.path.exists(root_path):
        os.mkdir(root_path)
    d = root_path+f"\\{keyword}\\"
    fo = os.path.exists(d)
    if not fo:
        os.mkdir(d)
        print("创建子目录成功")
    else:
        print("目录已经存在\n")
    return d
#下载图片到一个子目录
def download_pic(result_list,d):
    count = 1
    log_list = []
    for i in range(len(result_list)):
        try:
            all_path = d+str(i+1)+".jpg"
            with open(all_path,'wb') as f:
                f.write(requests.get(result_list[i]).content)
                log_list.append(f'第{count}张图片下载成功:  '+result_list[i])
                count += 1
        except:
            log_list.append(f'第{count}张图片ERROE:  '+result_list[i])
    return '\n'.join(log_list)
#定义类
class myThread(threading.Thread):
    def __init__(self,root_path,page,keyword):
        threading.Thread.__init__(self)
        self.root_path = root_path
        self.page = page
        self.keyword = keyword
    def run(self):
        encode_url_list = get_url_list(self.page,self.keyword)
        url_list = decode_url_list(encode_url_list)
        d = make_dir(self.root_path,self.keyword)
        log = download_pic(url_list,d)
        with open(self.root_path+"\\log.txt",'a+',encoding = 'utf-8') as f:
            f.write(log)
            f.write('\n\n\n———————————————————————\n')
#主函数
def main():
    start = time.time()
    root_path = os.getcwd()+"\\爬取百度图片"
    print(root_path)
    page,keyword_list = get_PagesAndKeyword()
    thread_pool = []
    for keyword in keyword_list:
        t = myThread(root_path,page,keyword)
        t.start()
        thread_pool.append(t)
    while thread_pool:
        t = thread_pool.pop()
        t.join()
    end = time.time()
    print(f"一共运行了{end-start}秒")
if __name__=="__main__":
    main()
